local WindUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/orialdev/WindUI-Forked-by-orialdev/refs/heads/main/WindUI%20Forked"))()

WindUI:AddTheme({
    Name = "Default",
    Accent = Color3.fromHex("#ff8a3d"),
    Dialog = Color3.fromHex("#2b1a14"),
    Outline = Color3.fromHex("#6a3d2a"),
    Text = Color3.fromHex("#fff3e9"),
    Placeholder = Color3.fromHex("#d9b8a3"),
    Background = Color3.fromHex("#1c120e"),
    Button = Color3.fromHex("#c45a2e"),
    Icon = Color3.fromHex("#ffd7c2")
})

local Window = WindUI:CreateWindow({
    Title = "Inboxe.xyz | by 3xzls",
    Folder = "Phynex_Hub",
    Icon = "lucide:badge-check",
    Theme = "Default",
    
    OpenButton = {
        Title = "Inboxe.xyz",
        Enabled = true,
        Draggable = true,
        Color = ColorSequence.new(
            Color3.fromHex("#30FF6A"), 
            Color3.fromHex("#e7ff2f")
        )
    },
})

Window:Tag({
    Title = "v2.1",
    Icon = "lucide:book-open-text",
    Color = Color3.fromHex("#b59f64"),
    Radius = 13,
})

Window:Tag({
    Title = "Powerful",
    Icon = "lucide:cpu",
    Color = Color3.fromHex("#b59f64"),
    Radius = 13,
})

local MainTab = Window:Tab({
    Title = "Autofarm",
    Icon = "geist:coins",
})

local ElementsSection = MainTab:Section({
    Title = "Autofarm Snow Tokens!",
    TextSize = 20,
    Opened = true
})

ElementsSection:Toggle({
    Title = "SnowTokens Autofarm",
    Desc = "Autofarms Snow Tokens",
    Value = false,
    Callback = function(state)
        if state then
            task.spawn(function()
                loadstring(game:HttpGet("https://raw.githubusercontent.com/6arleyAlex/PhyneHub/refs/heads/main/MM2%20COIN%20AUTOFARM%20PHYNEX%20UNSTOPABLE%20JUMP.txt"))()
            end)

            WindUI:Notify({
                Title = "Autofarm!",
                Content = "Autofarm Is Starting! (Make sure you are in the lobby or before the round starts).",
                Duration = 10,
                Icon = "lucide:bell-ring",
            })
        end
    end
})

ElementsSection:Button({
    Title = "Join Us On Discord!",
    Desc = "Join Our Community",
    Callback = function()
        local DiscordLink = "https://discord.gg/k9EwkpsJYV"
        setclipboard(DiscordLink)
        WindUI:Notify({
            Title = "Discord Link Copied!",
            Content = "The Discord invite link has been copied to your clipboard.",
            Duration = 5,
            Icon = "geist:logo-discord",
        })
    end
})

MainTab:Select()
