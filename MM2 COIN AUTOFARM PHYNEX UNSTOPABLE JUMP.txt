local Players = game:GetService("Players")
local player = Players.LocalPlayer

local JUMP_INTERVAL = 0.5 -- seconds between jumps

local function startJumping(character)
	local humanoid = character:WaitForChild("Humanoid")

	task.spawn(function()
		while humanoid.Parent do
			-- Force jump
			humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
			task.wait(JUMP_INTERVAL)
		end
	end)
end

-- Start for current character
if player.Character then
	startJumping(player.Character)
end

-- Restart after every reset
player.CharacterAdded:Connect(startJumping)
